/**
 * MERCADREAM - Central Engine v1.0
 * Unified State Management, Auth, and UI Sync
 */

const MercaDream = {
    state: {
        credits: 0,
        user: null,
        plan: 'free',
        lang: 'en'
    },

    init() {
        console.log("Initializing MercaDream Central Engine...");
        this.loadState();
        this.bindGlobalEvents();
        this.updateGlobalUI();
        this.checkAuth();
    },

    loadState() {
        this.state.credits = parseInt(localStorage.getItem('md_credits') || '0');
        this.state.plan = localStorage.getItem('md_plan') || 'free';
        this.state.lang = localStorage.getItem('md_lang') || 'en';
        try {
            this.state.user = JSON.parse(localStorage.getItem('md_user') || 'null');
        } catch (e) {
            this.state.user = null;
        }
    },

    saveState() {
        localStorage.setItem('md_credits', this.state.credits);
        localStorage.setItem('md_plan', this.state.plan);
        localStorage.setItem('md_lang', this.state.lang);
    },

    checkAuth() {
        const session = localStorage.getItem('sb_session');
        const isAuthPage = window.location.pathname.includes('login.html') || window.location.pathname.includes('register.html');
        
        if (!session && !isAuthPage && !window.location.pathname.endsWith('index.html')) {
            window.location.href = 'login.html';
        }
    },

    updateGlobalUI() {
        // Update all credit displays
        const formattedCredits = this.state.credits.toLocaleString();
        document.querySelectorAll('[data-credits], .nav-credits-val, #header-credits-val').forEach(el => {
            el.textContent = `${formattedCredits} CR`;
        });

        // Update user info
        if (this.state.user) {
            document.querySelectorAll('.user-name-display').forEach(el => {
                el.textContent = (this.state.user.displayName || this.state.user.email.split('@')[0]).toUpperCase();
            });
        }

        // Language specific UI adjustments
        document.documentElement.lang = this.state.lang;
        document.documentElement.dir = this.state.lang === 'ar' ? 'rtl' : 'ltr';
    },

    setCredits(value) {
        this.state.credits = value;
        this.saveState();
        this.updateGlobalUI();
    },

    deductCredits(amount) {
        if (this.state.credits >= amount) {
            this.setCredits(this.state.credits - amount);
            return true;
        }
        return false;
    },

    bindGlobalEvents() {
        // Sync across tabs
        window.addEventListener('storage', (e) => {
            if (['md_credits', 'md_plan', 'sb_session'].includes(e.key)) {
                this.loadState();
                this.updateGlobalUI();
                if (e.key === 'sb_session' && !e.newValue) window.location.href = 'index.html';
            }
        });

        // Shared navigation clicks
        document.querySelectorAll('[data-nav]').forEach(el => {
            el.onclick = (e) => {
                e.preventDefault();
                const target = el.getAttribute('data-nav');
                window.location.href = target.endsWith('.html') ? target : `${target}.html`;
            };
        });
    },

    logout() {
        ['sb_session', 'md_user', 'md_email', 'md_credits', 'md_plan', 'md_user_id'].forEach(k => localStorage.removeItem(k));
        window.location.href = 'index.html';
    }
};

// Auto-init on load
window.addEventListener('DOMContentLoaded', () => MercaDream.init());
